bash -c '
USB_PATH="$(find /media /run/media /mnt -type d -name "KAWSAR" -print -quit 2>/dev/null)"
if [ -z "$USB_PATH" ]; then
    echo "❌ USB KAWSAR not found! Plug in pendrive and try again."
    exit 1
fi
echo "✅ Found USB: $USB_PATH"
echo "📂 Installing Ollama..."
sudo cp "$USB_PATH/ollama" /usr/local/bin/ollama
sudo chmod +x /usr/local/bin/ollama
echo "📂 Copying AI model... please wait 2-3 minutes..."
cp "$USB_PATH/Qwen_Qwen3-8B-Q4_K_M.gguf" ~/
echo "📝 Creating config..."
cat > ~/Modelfile << EOF
FROM $HOME/Qwen_Qwen3-8B-Q4_K_M.gguf
PARAMETER num_ctx 8192
PARAMETER temperature 0.6
PARAMETER num_thread 12
EOF
echo "🚀 Starting AI service..."
sudo pkill ollama 2>/dev/null
sleep 3
/usr/local/bin/ollama serve &>/dev/null &
sleep 10
echo "🔧 Registering model..."
/usr/local/bin/ollama create qwen3 -f ~/Modelfile
rm -f ~/Modelfile
echo ""
echo "=========================================="
echo "✅ DONE! Unplug your USB."
echo "To start AI anytime type:"
echo "ollama run qwen3"
echo "=========================================="
'
